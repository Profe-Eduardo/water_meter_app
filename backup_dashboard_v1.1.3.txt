import 'dart:async';
import 'dart:convert';
import 'package:flutter/material.dart';
import 'package:http/http.dart' as http; // 🚀 Importante tener el paquete http

class DashboardScreen extends StatefulWidget {
  final double lecturaInicial;

  const DashboardScreen({super.key, required this.lecturaInicial});

  @override
  State<DashboardScreen> createState() => _DashboardScreenState();
}

class _DashboardScreenState extends State<DashboardScreen> {
  // Variables de control reales
  late double litrosConsumidos;
  double flujoPorMinuto = 0.0; // 🚀 Nueva métrica real
  String estadoSensor = "Buscando...";
  bool conectado = false;

  Timer? _timerMonitoreo;
  // Dirección IP del ESP32 en modo Access Point (ajusta si usas otra)
  final String _esp32Url = 'http://192.168.4.1/datos';

  @override
  void initState() {
    super.initState();
    litrosConsumidos = widget.lecturaInicial;

    // 🚀 Iniciamos la consulta en tiempo real automática (Cada 1 segundo)
    _iniciarMonitoreoReal();
  }

  @override
  void dispose() {
    _timerMonitoreo?.cancel(); // Limpieza obligatoria de memoria
    super.dispose();
  }

  // 🔄 FUNCIÓN MOTOR: Consulta al hardware real por HTTP
  void _iniciarMonitoreoReal() {
    _timerMonitoreo = Timer.periodic(const Duration(seconds: 1), (timer) async {
      try {
        final response = await http
            .get(Uri.parse(_esp32Url))
            .timeout(
              const Duration(
                seconds: 2,
              ), // Si tarda más de 2s, asumimos desconexión
            );

        if (response.statusCode == 200) {
          final data = jsonDecode(response.body);
          setState(() {
            conectado = true;
            estadoSensor = "Conectado";
            // Tomamos el acumulado del ESP32 y le sumamos la lectura inicial si aplica
            litrosConsumidos =
                widget.lecturaInicial +
                (data['litros_totales'] as num).toDouble();
            flujoPorMinuto = (data['flujo_por_minuto'] as num).toDouble();
          });
        }
      } catch (e) {
        // Si el chip se apaga o el Wi-Fi se desconecta, cae aquí inmediatamente
        setState(() {
          conectado = false;
          estadoSensor = "Desconectado";
          flujoPorMinuto = 0.0;
        });
      }
    });
  }

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(
        title: const Text(
          'Monitoreo Water Meter',
          style: TextStyle(color: Colors.white, fontWeight: FontWeight.bold),
        ),
        backgroundColor: conectado
            ? Colors.teal
            : Colors.blueGrey, // 🚀 Cambia de color según red
        centerTitle: true,
        actions: [
          IconButton(
            icon: Icon(
              conectado ? Icons.wifi : Icons.wifi_off,
              color: conectado ? Colors.white : Colors.redAccent,
            ),
            onPressed: () {},
          ),
        ],
      ),
      body: Column(
        children: [
          // 🚨 ALERTA VISUAL: Aparece arriba solo si el ESP32 se desconecta
          if (!conectado)
            Container(
              width: double.infinity,
              color: Colors.redAccent,
              padding: const EdgeInsets.symmetric(vertical: 8.0),
              child: const Row(
                mainAxisAlignment: MainAxisAlignment.center,
                children: [
                  SizedBox(
                    width: 16,
                    height: 16,
                    child: CircularProgressIndicator(
                      strokeWidth: 2,
                      color: Colors.white,
                    ),
                  ),
                  SizedBox(width: 12),
                  Text(
                    'Buscando señal del medidor... Verifica tu Wi-Fi',
                    style: TextStyle(
                      color: Colors.white,
                      fontWeight: FontWeight.bold,
                    ),
                  ),
                ],
              ),
            ),

          // 📱/📑 DISEÑO ADAPTATIVO (Celular / Tablet)
          Expanded(
            child: LayoutBuilder(
              builder: (context, constraints) {
                // Si el ancho de pantalla es mayor a 600px se ejecuta el modo Tablet
                if (constraints.maxWidth > 600) {
                  return _buildTabletLayout();
                } else {
                  return _buildMobileLayout();
                }
              },
            ),
          ),
        ],
      ),
    );
  }

  // 📱 DISEÑO VERTICAL PARA CELULARES
  Widget _buildMobileLayout() {
    return SingleChildScrollView(
      padding: const EdgeInsets.all(16.0),
      child: Column(
        crossAxisAlignment: CrossAxisAlignment.stretch,
        children: [
          _buildTarjetaEstado(),
          const SizedBox(height: 16),
          _buildTarjetaConsumo(),
          const SizedBox(height: 16),
          _buildTarjetaFlujo(), // Nueva métrica agregada
        ],
      ),
    );
  }

  // 📑 DISEÑO EN DOS COLUMNAS PARA TABLETS
  Widget _buildTabletLayout() {
    return Padding(
      padding: const EdgeInsets.all(24.0),
      child: Row(
        crossAxisAlignment: CrossAxisAlignment.start,
        children: [
          // Columna Izquierda: Estado del hardware
          Expanded(
            flex: 4,
            child: Column(
              children: [
                _buildTarjetaEstado(),
                const SizedBox(height: 16),
                _buildTarjetaFlujo(),
              ],
            ),
          ),
          const SizedBox(width: 20),
          // Columna Derecha: Lectura central grande
          Expanded(flex: 6, child: _buildTarjetaConsumo()),
        ],
      ),
    );
  }

  // --- COMPONENTES ORIGINALES PULIDOS ---

  Widget _buildTarjetaEstado() {
    return Card(
      elevation: 4,
      shape: RoundedRectangleBorder(borderRadius: BorderRadius.circular(12)),
      child: Padding(
        padding: const EdgeInsets.all(16.0),
        child: Row(
          children: [
            Icon(
              Icons.router,
              size: 40,
              color: conectado ? Colors.teal : Colors.blueGrey,
            ),
            const SizedBox(width: 16),
            Column(
              crossAxisAlignment: CrossAxisAlignment.start,
              children: [
                const Text(
                  'Dispositivo: ESP32_Server',
                  style: TextStyle(fontSize: 14, color: Colors.grey),
                ),
                Text(
                  'Estado: $estadoSensor',
                  style: TextStyle(
                    fontSize: 18,
                    fontWeight: FontWeight.bold,
                    color: conectado ? Colors.green : Colors.red,
                  ),
                ),
              ],
            ),
          ],
        ),
      ),
    );
  }

  Widget _buildTarjetaConsumo() {
    return Card(
      elevation: 6,
      color: conectado ? Colors.teal.shade50 : Colors.grey.shade100,
      shape: RoundedRectangleBorder(borderRadius: BorderRadius.circular(15)),
      child: Padding(
        padding: const EdgeInsets.symmetric(vertical: 32.0, horizontal: 16.0),
        child: Column(
          mainAxisAlignment: MainAxisAlignment.center,
          children: [
            Icon(
              Icons.opacity,
              size: 60,
              color: conectado ? Colors.blue : Colors.grey,
            ),
            const SizedBox(height: 10),
            Text(
              'CONSUMO TOTAL',
              style: TextStyle(
                fontSize: 16,
                fontWeight: FontWeight.bold,
                color: conectado ? Colors.teal : Colors.blueGrey,
              ),
            ),
            const SizedBox(height: 10),
            Text(
              '${litrosConsumidos.toStringAsFixed(1)} L',
              style: TextStyle(
                fontSize: 48,
                fontWeight: FontWeight.bold,
                color: conectado ? Colors.teal.shade900 : Colors.black54,
              ),
            ),
            const SizedBox(height: 5),
            Text(
              'Partiendo de la lectura inicial: ${widget.lecturaInicial.toStringAsFixed(1)} L',
              style: const TextStyle(
                fontSize: 12,
                color: Colors.grey,
                fontStyle: FontStyle.italic,
              ),
            ),
          ],
        ),
      ),
    );
  }

  Widget _buildTarjetaFlujo() {
    return Card(
      elevation: 4,
      shape: RoundedRectangleBorder(borderRadius: BorderRadius.circular(12)),
      child: Padding(
        padding: const EdgeInsets.all(16.0),
        child: Row(
          children: [
            Icon(
              Icons.speed,
              size: 40,
              color: flujoPorMinuto > 0 ? Colors.blue : Colors.grey,
            ),
            const SizedBox(width: 16),
            Column(
              crossAxisAlignment: CrossAxisAlignment.start,
              children: [
                const Text(
                  'Velocidad de Flujo',
                  style: TextStyle(fontSize: 14, color: Colors.grey),
                ),
                Text(
                  '${flujoPorMinuto.toStringAsFixed(2)} L/min',
                  style: const TextStyle(
                    fontSize: 18,
                    fontWeight: FontWeight.bold,
                    color: Colors.blue,
                  ),
                ),
              ],
            ),
          ],
        ),
      ),
    );
  }
}
